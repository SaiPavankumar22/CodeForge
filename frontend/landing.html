two_sum.py ✓ Accepted 
def twoSum(nums, target):
    seen = {}
    for i, n in enumerate(nums):
        complement = target - n
        if complement in seen:
            return [seen[complement], i]
        seen[n] = i
# Time: O(n)  Space: O(n)
